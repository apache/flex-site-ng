Dependency Information

Apache Maven

<dependency>
  <groupId>org.apache.flex.website</groupId>
  <artifactId>flex-site</artifactId>
  <version>0.8.0-SNAPSHOT</version>
  <type>pom</type>
</dependency>

Apache Buildr

'org.apache.flex.website:flex-site:pom:0.8.0-SNAPSHOT'

Apache Ivy

<dependency org="org.apache.flex.website" name="flex-site" rev="0.8.0-SNAPSHOT">
  <artifact name="flex-site" type="pom" />
</dependency>

Groovy Grape

@Grapes(
@Grab(group='org.apache.flex.website', module='flex-site', version='0.8.0-SNAPSHOT')
)

Gradle/Grails

compile 'org.apache.flex.website:flex-site:0.8.0-SNAPSHOT'

Scala SBT

libraryDependencies += "org.apache.flex.website" % "flex-site" % "0.8.0-SNAPSHOT"

Leiningen

[org.apache.flex.website/flex-site "0.8.0-SNAPSHOT"]
